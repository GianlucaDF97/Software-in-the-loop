107    



108else if ((H_data_to_AIR_RX[2] == 216) && (H_data_to_AIR_RX[3] == 4)) { // è la condizione che si verifica quando si attiva la modalità INTERVENTO PILOTA

112}



113



114// --- fine parte aggiunta --- //



115



116if (count[0] == 0) //inizializzazione



117{



118    mavlink_status_t* status = mavlink_get_channel_status(MAVLINK_COMM_0);



119    memset(status, 0, sizeof(mavlink_status_t));



120}



121



122if (mem_in[7]!=count[0]) //controllo che è arrivato un messaggio nuovo



123    



124{



125    int len = len_in[0]; //lunghezza buffer



126    // Inserisce i bytes, uno alla volta, nel buffer



127    for (i=0; i<len; i++)



128    {



129        uint8_T c = (uint8_T)bytes_in[i];



130        uint8_t msg_received = mavlink_parse_char(MAVLINK_COMM_0, c, &msg, &status_copy);



131        



132        if (msg_received) // se ho un msessaggio intero



133        {



134            // mem_out e mem_in contengono le informazioni della tipologia di messaggio



135            mem_out[0]=msg.sysid;    // chi interroga



136            mem_out[1]=msg.compid;   // chi interroga

138            mem_out[3]=msg.len;      // lunghezza del tipo di messaggio



139            mem_out[4]=msg.seq;     // messaggio numero # spedito dalla stazione di terra



140            mem_out[6]=0;             // led per verifiche



141            // per calcolare messaggi persi



142            if (mem_out[4]>mem_in[4]) // poi deve per forza essere mem_out[4]>mem_in[4]



143            {



144                if (mem_out[4]!=(mem_in[4]+1))



145                    mem_out[5]=mem_in[5]+(mem_out[4]-mem_in[4]); // numero messaggi persi



146                    //mem_out[5]=0;



147            }



148            else // altrimenti il count si è resettato e quindi resetto anche il conteggio dei messaggi persi



149                mem_out[5]=0;



150



151            switch (msg.msgid)      // controllo che tipo di messaggio sta arrivando



152            {



153                case 0: // HEARTBEAT



154                {



155                    mavlink_heartbeat_t heart255;  // struttura heartbeat



156                    mavlink_msg_heartbeat_decode(&msg, &heart255); // preleva i valori in arrivo e li mette nella struttura



157                    Heartbeat_data[0]=heart255.type;



158                    Heartbeat_data[1]=heart255.autopilot;



159                    Heartbeat_data[2]=heart255.base_mode;



160                    Heartbeat_data[3]=heart255.system_status;



161                    Heartbeat_custom[0]=heart255.custom_mode;



162                    break;



163                }



164                



165                case 76:  // COMMAND_LONG



166                {       // mavlink_set_mode_t setmode255;  // struttura setmode



167//                         mavlink_msg_set_mode_decode  (&msg, &setmode255);  //preleva i valori in arrivo e li mette nella struttura



168//                         if   (setmode255.target_system==AP_SI[0])//pari al SI del UAV !!ATTENZIONE!!



169//                         {



170//                             Base_mode[0]=setmode255.base_mode;



171//                         }



172                        mavlink_command_long_t cmd_long255;



173                        mavlink_msg_command_long_decode(&msg, &cmd_long255);



174                        



175                        if (cmd_long255.command == 176) // 176 è l'id di CMD_DO_SET_MODE



176                        {



177                            Base_mode[0] = cmd_long255.param1; // Base_mode è utilizzato nella successiva Matlab Function e rappresenta il MAV_MODE 



178                            Mav_state[0] = cmd_long255.param3; // serve per specificare il MAV_STATE



179                            



180                            int_sw_safe_interv[0] = 0; // ogni qual volta si setta una modalità di volo l'interruttore del safety switch è posto a zero



181                            int_sw_safe_interv[1] = 0; // ogni qual volta si setta una modalità di volo l'interruttore dell'intervento pilota è posto a zero



182                        }



183                        



184                        // Parte aggiunta da sensolini per l'invio della pressione pista prima dell'atterraggio



185                        



186                        if (cmd_long255.command == 113) // 113 è l'id del messaggio usato



187                        {



188                            int P_param[7] = {0};



189                            P_param[0] = cmd_long255.param1; 



190                            P_param[1] = cmd_long255.param2;



191                            P_param[2] = cmd_long255.param3;



192                            P_param[3] = cmd_long255.param4;



193                            P_param[4] = cmd_long255.param5;



194                            P_param[5] = cmd_long255.param6;



195

197                            



198                            if (P_ref[0] != 101325)

200                             P_int[0] = 1;



201                            }



202                          



203                        }

205                        // Parte aggiunta per l'operazione di calibrazione dell'altimetro e test del safe e dell'intervento pilota durante la fase di pre-flight



206                       



207                        if (H_mav[1] == 2) { // se è inviato un system_status = 2 che imposta il MAV_STATE come MAV_STATE_CALIBRATING



208                            



209                            if (cmd_long255.command == 241) { // 241 è l'id di MAV_CMD_PREFLIGHT_CALIBRATION



210                        



211                                Cal_alt[0] = cmd_long255.param3; // param3 sarà pari a 1 per la calibrazione dell'altimetro (ground pressure)                              



212                            }



213                                               



214                            if (cmd_long255.command == 180) { // 180 è l'id di MAV_CMD_DO_SET_PARAMETER



215                        



216                                T_number[0] = 13; // è la componente riferita alla temperatura di riferimento dell'altimetro del vettore riferimenti presente nella successiva Matlab Function



217                                T_int[0]    = cmd_long255.param1; // funge da interruttore e sarà pari a 1



218                                T_ref[0]    = cmd_long255.param2; // param2 è il valore della T di riferimento da impostare da QGC



219                            }



220							



221							if (cmd_long255.command == 179){ // 179 è l'id di MAV_CMD_DO_SET_HOME



222                        



223                                Cal_home[0] = cmd_long255.param1;



224								Cal_home[1] = cmd_long255.param2;



225                                Cal_home[2] = cmd_long255.param3;



226                            }



227                        }



228                        



229                        if (H_mav[1] == 1) { // se è inviato un system_status = 1 che imposta il MAV_STATE come MAV_STATE_BOOT



230                            



231                            if (cmd_long255.command == 180) { // 180 è l'id di MAV_CMD_DO_SET_PARAMETER



232                                



233                                int_sw_safe_interv[0] = cmd_long255.param1; // se voglio testare il safe imposto param1 = 1 e param2 = 0

236                        }

257                    mavlink_param_set_t paramset255; // struttura di param_set

276                            GC_request[1]=4; // I_RC



277                        else if (paramset255.param_id[0]==73  && paramset255.param_id[1]==95 &&paramset255.param_id[2]==87 && paramset255.param_id[3]==80)



278                            GC_request[1]=5; // I_WP



279                        else if (paramset255.param_id[0]==73  && paramset255.param_id[1]==95 &&paramset255.param_id[2]==65 && paramset255.param_id[3]==84 && paramset255.param_id[4]==79)



280                            GC_request[1]=6; // I_ATO



281                         else if (paramset255.param_id[0]==73  && paramset255.param_id[1]==95 &&paramset255.param_id[2]==65 && paramset255.param_id[3]==76)



282                            GC_request[1]=7; // I_AL



283                        



284                        else if (paramset255.param_id[0]==86  && paramset255.param_id[1]==95 && paramset255.param_id[2]==72 && paramset255.param_id[3]==68 && paramset255.param_id[4]==71)



285                            GC_request[1]=8; // V_HDG



286                        else if (paramset255.param_id[0]==86  && paramset255.param_id[1]==95 &&paramset255.param_id[2]==73 && paramset255.param_id[3]==65 && paramset255.param_id[4]==83)

289                            GC_request[1]=10; // V_ALT



290                        else if (paramset255.param_id[0]==86  && paramset255.param_id[1]==95 &&paramset255.param_id[2]==82 && paramset255.param_id[3]==67)



291                            GC_request[1]=11; // V_RC



292                        



293                        else



294                            GC_request[1]=255; // arriva una richiesta ma non la riconosco e quindi permetto l'invio di tutti i parametri

296                    break;



297                }

378                default:

383        } // chiusra msg_received

390} // fine count
